直接用java的类Process 和 Runtime实现，代码。。  
       
  
Process process = null;  
  
try{  
  
process = Runtime.getRuntime().exec("su");  
  
//这里是主要程序代码   
  
process.waitFor();  
  
}catch(Exception e){  
  
e.printStackTrace();  
  
}finally {  
  
process.destroy();  
  
}
